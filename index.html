Quick start
  1. 1Download the belvilla_scraper ZIP from Claude (4 Python files)
  2. 2Install Python 3.10+ if not already installed (python.org), then run:
    pip install -r requirements.txt
  3. 3Copy .env.example → .env and paste in your API keys (see table)
  4. 4Run a one-shot scrape:
    python scraper.py --regions all --limit 30
  5. 5Schedule daily runs:
    python scheduler.py --interval 24h
  6. 6Drop the output leads.csv into the Import CSV tab — done!

Apps Script code — paste this into Extensions → Apps Script
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

    if (data.action === 'replace') {
      // Keep header row, clear everything else
      var lastRow = sheet.getLastRow();
      if (lastRow > 1) sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).clearContent();

      // Write all leads
      var rows = data.leads.map(function(l) {
        return [
          l.name||'', l.email||'', l.phone||'', l.country||'',
          l.region||'', l.platform||'', l.properties_est||1,
          l.url||'', l.website||'', l.source_url||'',
          l.status||'New', l.notes||'', l.added||''
        ];
      });
      if (rows.length > 0) {
        sheet.getRange(2, 1, rows.length, rows[0].length).setValues(rows);
      }
    }

    return ContentService.createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false, error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService.createTextOutput('Belvilla Leads Script is running.');
